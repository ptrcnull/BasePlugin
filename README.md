```xml
<repository>
    <id>ptrcnull</id>
    <url>https://maven.ptrcnull.me/repository/maven-releases/</url>
</repository>

<dependencies>
    <dependency>
        <groupId>me.ptrcnull</groupId>
        <artifactId>ConfigurablePlugin</artifactId>
        <version>1.0.0</version>
    </dependency>
</dependencies>
```
